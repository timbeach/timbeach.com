# WASM Neural TTS Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a neural WASM TTS engine (Kokoro) to the site's read-aloud feature so it works on every browser without system dependencies, while preserving the existing Web Speech API as the zero-cost default.

**Architecture:** Introduce an **engine interface** that both the existing Web Speech path (`nativeEngine`) and the new Kokoro WASM path (`kokoroEngine`) implement. A **playback controller** above both owns the paragraph queue and look-ahead pipeline. A new "HQ" button in the TTS bar swaps the active engine on user request.

**Tech Stack:** Vanilla JS, `kokoro-js@1.2.0` (dynamic ESM import from jsDelivr), Kokoro-82M-v1.0-ONNX (q8 quantization, ~80 MB, HuggingFace primary + VPS-hosted local fallback). No build step.

**Spec:** `docs/superpowers/specs/2026-04-21-wasm-tts-design.md`

```javascript
    // =========================================================
    // Engine interface: { name, init(), voices, synth(text, voiceId, rate), cancel() }
    // =========================================================

    const nativeEngine = {
      name: 'native',
      voices: [],
      _synth: window.speechSynthesis,
      _loaded: false,

      async init() {
        // Voices may populate synchronously (Safari/Firefox) or via onvoiceschanged (Chromium).
        // Wait up to 1 second for voices to appear.
        const loadOnce = () => {
          const raw = this._synth.getVoices();
          if (!raw.length) return false;
          const english = raw.filter(v => v.lang.startsWith('en'));
          const sorted = english.sort((a, b) => {
            const an = /natural|neural|enhanced|premium/i.test(a.name) ? 0 : 1;
            const bn = /natural|neural|enhanced|premium/i.test(b.name) ? 0 : 1;
            return an - bn || a.name.localeCompare(b.name);
          });
          const others = raw.filter(v => !v.lang.startsWith('en')).slice(0, 5);
          this.voices = [...sorted, ...others].map(v => ({
            id: v.name,
            label: `${v.name.replace(/^(Microsoft |Google |Apple )/, '')} (${v.lang})`,
            lang: v.lang,
            _raw: v,
          }));
          this._loaded = true;
          return true;
        };

        if (loadOnce()) return;

        return new Promise(resolve => {
          const handler = () => { loadOnce(); resolve(); };
          if (this._synth.onvoiceschanged !== undefined) {
            this._synth.onvoiceschanged = handler;
          }
          // Hard timeout: 1 second. Callers use `this.voices.length === 0` to detect no-voices state.
          setTimeout(() => resolve(), 1000);
        });
      },

      async synth(text, voiceId /* string */, rate /* number */) {
        // Returns a pseudo-HTMLAudioElement: {play, pause, playbackRate, addEventListener('ended', fn)}
        // Internally uses SpeechSynthesisUtterance.
        const utterance = new SpeechSynthesisUtterance(text);
        utterance.rate = rate;
        const voice = this.voices.find(v => v.id === voiceId);
        if (voice) utterance.voice = voice._raw;

        let endCallbacks = [];
        let started = false;
        let cancelled = false;

        const handle = {
          _utterance: utterance,
          _cancelled: false,
          playbackRate: rate,
          play() {
            if (this._cancelled) return;
            nativeEngine._synth.cancel();  // clear stale state (Chromium bug)
            setTimeout(() => {
              if (this._cancelled) return;
              nativeEngine._synth.speak(utterance);
            }, 50);
          },
          pause() { nativeEngine._synth.pause(); },
          resume() { nativeEngine._synth.resume(); },
          addEventListener(evt, cb) {
            if (evt === 'ended') endCallbacks.push(cb);
          },
          _cancel() {
            this._cancelled = true;
            nativeEngine._synth.cancel();
          },
        };

        utterance.onstart = () => { started = true; };
        utterance.onend = () => {
          // Chromium bug: onend can fire instantly without actually speaking.
          // If onend fires before onstart and we're not cancelled, retry once.
          if (!started && !handle._cancelled) {
            setTimeout(() => {
              if (!handle._cancelled) nativeEngine._synth.speak(utterance);
            }, 50);
            return;
          }
          endCallbacks.forEach(cb => cb());
        };
        utterance.onerror = (e) => {
          if (e.error === 'interrupted' || e.error === 'canceled') return;
          endCallbacks.forEach(cb => cb());  // treat errors as end, keep reading
        };

        return handle;
      },

      cancel() {
        this._synth.cancel();
      },
    };
```

```javascript
    function ttsSynth(idx, text) {
      if (tts.queue.inflight.has(idx)) {
        return tts.queue.inflight.get(idx).promise;
      }
      const abort = { cancelled: false };
      const promise = tts.engine.synth(text, tts.selectedVoiceId, tts.rate)
        .finally(() => tts.queue.inflight.delete(idx));
      tts.queue.inflight.set(idx, { promise, abort });
      return promise;
    }
```

Insert after the `nativeEngine` closing `};`:

```javascript
    const kokoroEngine = {
      name: 'kokoro',
      voices: [
        { id: 'af_bella',   label: 'Bella (US)',    lang: 'en-US' },
        { id: 'am_michael', label: 'Michael (US)',  lang: 'en-US' },
        { id: 'bm_george',  label: 'George (UK)',   lang: 'en-GB' },
      ],
      _tts: null,         // KokoroTTS instance
      _loaded: false,
      _abort: null,       // AbortController for in-flight loads

      async init(onProgress) {
        if (this._loaded) return;
        // Placeholder: Task 6 fills in the real download logic.
        throw new Error('kokoroEngine.init not implemented yet');
      },

      async synth(text, voiceId, rate) {
        if (!this._loaded) throw new Error('kokoroEngine not initialized');
        // Kokoro generates PCM; we wrap it in a WAV blob and return an HTMLAudioElement.
        const { audio, sampling_rate } = await this._tts.generate(text, {
          voice: voiceId || 'af_bella',
          speed: 1.0,  // we use audio.playbackRate instead, for uniformity with nativeEngine
        });
        const wav = kokoroEngine._encodeWav(audio, sampling_rate);
        const blob = new Blob([wav], { type: 'audio/wav' });
        const url = URL.createObjectURL(blob);
        const el = new Audio(url);
        el._blobUrl = url;  // so ttsStop can revoke
        el.playbackRate = rate;
        return el;
      },

      cancel() {
        if (this._abort) this._abort.abort();
      },

      // Minimal WAV encoder: Float32Array PCM -> WAV ArrayBuffer.
      _encodeWav(samples, sampleRate) {
        const numChannels = 1;
        const bytesPerSample = 2;  // int16
        const buffer = new ArrayBuffer(44 + samples.length * bytesPerSample);
        const view = new DataView(buffer);
        const writeStr = (offset, str) => {
          for (let i = 0; i < str.length; i++) view.setUint8(offset + i, str.charCodeAt(i));
        };
        writeStr(0, 'RIFF');
        view.setUint32(4, 36 + samples.length * bytesPerSample, true);
        writeStr(8, 'WAVE');
        writeStr(12, 'fmt ');
        view.setUint32(16, 16, true);
        view.setUint16(20, 1, true);  // PCM
        view.setUint16(22, numChannels, true);
        view.setUint32(24, sampleRate, true);
        view.setUint32(28, sampleRate * numChannels * bytesPerSample, true);
        view.setUint16(32, numChannels * bytesPerSample, true);
        view.setUint16(34, 16, true);  // bits per sample
        writeStr(36, 'data');
        view.setUint32(40, samples.length * bytesPerSample, true);
        let offset = 44;
        for (let i = 0; i < samples.length; i++) {
          const s = Math.max(-1, Math.min(1, samples[i]));
          view.setInt16(offset, s < 0 ? s * 0x8000 : s * 0x7fff, true);
          offset += 2;
        }
        return buffer;
      },
    };
```

## Task 9: Look-ahead pipeline

**Files:**
- Modify: `index.html` (TTS script block, `ttsSpeakCurrent`)

**Goal:** Start synthesizing paragraph `i+1` in parallel with playback of paragraph `i`, so there's no perceived wait between paragraphs after the first.

- [ ] **Step 1: Rewrite `ttsSpeakCurrent()` with look-ahead**

Replace the entire body of `ttsSpeakCurrent()`:

```javascript
    async function ttsSpeakCurrent() {
      if (!tts.isPlaying || tts.currentIndex < 0) return;
      if (tts.currentIndex >= tts.paragraphs.length) {
        ttsStop();
        tts.statusText.textContent = 'finished';
        return;
      }

      const idx = tts.currentIndex;
      const para = tts.paragraphs[idx];
      const text = ttsGetText(para);
      if (!text) {
        tts.currentIndex++;
        ttsSpeakCurrent();
        return;
      }

      tts.paragraphs.forEach(p => p.classList.remove('tts-reading'));
      para.classList.add('tts-reading');
      tts.statusText.textContent = `reading paragraph ${idx + 1} of ${tts.paragraphs.length}`;
      ttsUpdateProgress();

      // Use preexisting audio from queue if this is an advance, else synthesize
      let audio;
      if (tts.queue.next && tts.queue.next.idx === idx) {
        audio = tts.queue.next.audio;
        tts.queue.next = null;
      } else {
        // First paragraph or cold start
        const synthStartMs = Date.now();
        const synthPromise = ttsSynth(idx, text);
        // Show "synthesizing..." if synth takes longer than 500ms
        const loadingTimer = setTimeout(() => {
          tts.statusText.textContent = 'synthesizing…';
        }, 500);
        try {
          audio = await synthPromise;
        } finally {
          clearTimeout(loadingTimer);
        }
        if (!tts.isPlaying || tts.currentIndex !== idx) return;
      }

      tts.queue.current = { idx, audio, text };
      tts._currentAudio = audio;
      audio.playbackRate = tts.rate;

      // Kick off look-ahead for idx+1 (if exists and not already queued)
      const nextIdx = idx + 1;
      if (nextIdx < tts.paragraphs.length && !tts.queue.next) {
        const nextText = ttsGetText(tts.paragraphs[nextIdx]);
        if (nextText) {
          ttsSynth(nextIdx, nextText).then((nextAudio) => {
            if (tts.currentIndex === idx) {  // still current, stash it
              tts.queue.next = { idx: nextIdx, audio: nextAudio, text: nextText };
            } else {
              // We already advanced past; discard
              if (nextAudio._blobUrl) URL.revokeObjectURL(nextAudio._blobUrl);
            }
          }).catch((e) => {
            console.warn('[tts] look-ahead synth failed for idx', nextIdx, e);
          });
        }
      }

      audio.addEventListener('ended', () => {
        if (!tts.isPlaying || tts._currentAudio !== audio) return;
        tts.statusText.textContent = `reading paragraph ${idx + 1} of ${tts.paragraphs.length}`;
        // If the next paragraph's audio isn't ready yet, show "buffering next…"
        if (!tts.queue.next || tts.queue.next.idx !== idx + 1) {
          if (idx + 1 < tts.paragraphs.length) {
            tts.statusText.textContent = 'buffering next…';
          }
        }
        // Revoke this paragraph's blob URL (memory hygiene)
        if (audio._blobUrl) URL.revokeObjectURL(audio._blobUrl);
        tts.currentIndex++;
        ttsSpeakCurrent();
      });

      audio.play();
    }
```
